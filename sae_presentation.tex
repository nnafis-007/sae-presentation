\begin{frame}{Superposition in LLMs}
\textbf{Core issue}
\begin{itemize}
  \item LLM activations can encode more independent features than available neurons
  \item A single neuron may respond to multiple unrelated concepts
  \item This appears as \textbf{polysemanticity}, a signature of superposition
\end{itemize}

\vspace{0.4em}
\textbf{Consequence}
\begin{itemize}
  \item Neuron-level interpretation becomes unstable and ambiguous
  \item Hard to assign one clean concept to one neuron
\end{itemize}
\end{frame}

\begin{frame}{Superposition Intuition}
\textbf{Geometric view}
\begin{itemize}
  \item Latent space has dimension $d$
  \item Model needs to represent $c$ concepts with $c \gg d$
  \item Concepts are packed as overlapping linear combinations
\end{itemize}

Observed activation vector:
\[
z \approx \sum_{i=1}^{c} a_i f_i
\]
\begin{itemize}
  \item $f_i$: feature directions
  \item $a_i$: feature strengths
\end{itemize}
\end{frame}

\slideimg{Usual_LLM_superposition.png}
\end{frame}

\begin{frame}{Why This Hurts Interpretability}
\begin{itemize}
  \item Raw neuron activations are not aligned with human concepts
  \item A neuron can fire for different causes in different contexts
  \item Causal interventions at neuron level can be noisy
\end{itemize}

\vspace{0.5em}
\textbf{Goal of SAE-based Interpretibility}
\begin{itemize}
  \item Decompose dense/polysemantic representations
  \item Recover sparse, more monosemantic feature activations
\end{itemize}
\end{frame}

\begin{frame}{How Sparsity Solves Superposition}
\textbf{Key idea: sparse dictionary coding}

Learn:
\begin{itemize}
  \item an overcomplete dictionary ($m > d$)
  \item sparse coefficients so only a few features are active per token
\end{itemize}

Then each input is reconstructed by a small active subset:
\[
z \rightarrow h(z) \rightarrow \hat{z}
\]

\begin{itemize}
  \item Sparsity encourages feature specialization
  \item Specialization improves interpretability and causal tracing
\end{itemize}
\end{frame}

\begin{frame}{SAE Intro: Architecture}
\begin{columns}[T,onlytextwidth]
\column{0.45\textwidth}
Standard SAE components:
\begin{enumerate}
  \item \textbf{Encoder}: $z \in \mathbb{R}^d \rightarrow h(z) \in \mathbb{R}^m$
  \item \textbf{Decoder}: reconstruct $\hat{z}$ from sparse code
\end{enumerate}

Typical form:
\[
\hat{z} = h(z) W_{dec} + b_{dec}
\]

\begin{itemize}
  \item Overcomplete regime: $m \gg d$
  \item Sparse features are easier to interpret
\end{itemize}

\end{columns}
\end{frame}

\begin{frame}{Sparse Representation Principle}
\begin{columns}[T,onlytextwidth]
\column{0.52\textwidth}
Given encoder output $h(z)$:
\begin{itemize}
  \item Most entries in $h(z)$ should be 0 (or near 0)
  \item Only informative features should activate
  \item This pushes the model away from "everything somewhat active"
  \item Toward "few meaningful features strongly active"
\end{itemize}

Result: cleaner feature-to-concept mapping.

\column{0.46\textwidth}
\begin{center}

\end{center}
\end{columns}
\end{frame}

\begin{frame}{SAE Loss (L1-Regularized Form)}
From the survey's standard objective:
\[
L(z) = \|z - \hat{z}\|_2^2 + \alpha \|h(z)\|_1
\]

\begin{itemize}
  \item Reconstruction term keeps information fidelity
  \item $L_1$ term enforces sparse activations
  \item $\alpha$ controls the trade-off
\end{itemize}

\textbf{Trade-off}
\begin{itemize}
  \item High $\alpha$: more sparse, weaker reconstruction
  \item Low $\alpha$: better reconstruction, less disentanglement
\end{itemize}
\end{frame}

\begin{frame}{Common Failure Cases}
\begin{itemize}
  \item Dead features: never selected under current data and hyperparameters
  \item Split concepts: one concept fragmented across many weak features
  \item Polysemantic leftovers: one feature still mixes unrelated contexts
  \item Dataset bias: labels may reflect corpus skew
\end{itemize}
\end{frame}


\begin{frame}{Tuning Parameters (Practical)}
\textbf{Core knobs}
\begin{itemize}
  \item Dictionary size $m$ (expansion factor)
  \item Sparsity strength: $\alpha$ (L1 regularization)
  \item Learning rate and optimizer settings
  \item Batch size and token sampling strategy
\end{itemize}

\textbf{Typical tuning workflow}
\begin{enumerate}
  \item Start with stable reconstruction
  \item Increase sparsity gradually
  \item Track dead features and usage spread
  \item Select checkpoint with best interpretability/reconstruction balance
\end{enumerate}
\end{frame}

\begin{frame}{Evolution of SAE in Practice}
The survey groups SAE evolution into two broad directions:
\begin{itemize}
  \item \textbf{Architecture improvements}: Gated, TopK, JumpReLU, Feature Choice, etc.
  \item \textbf{Training/evaluation improvements}: better sparsity control, dead-feature mitigation, stronger diagnostics
\end{itemize}

\begin{block}{Practical takeaway}
\vspace{2pt}
Do not optimize only reconstruction loss; monitor sparsity quality and feature utilization jointly.
\end{block}
\end{frame}

\begin{frame}{FC\_SAE (Feature Choice SAE)}
Feature Choice SAE changes sparsity allocation direction.


\begin{itemize}
  \item Instead of fixed active features per token
  \item It enforces each feature gets assigned to a controlled number of tokens
\end{itemize}

(Ayonrinde, 2024):
\[
\sum_j S_{i,j} = m,\ \forall i
\]

\begin{itemize}
  \item $S_{i,j}$ indicates whether feature $i$ is active for token $j$
  \item Promotes more uniform feature utilization
  \item Targets dead-feature collapse directly
\end{itemize}
\end{frame}

\vspace{0.3em}
\normalsize
\textbf{Interpretation:}
\begin{itemize}
  \item openwebtext shows much higher dead-feature collapse than peoples\_speech (55.21\% vs 38.38\%).
  \item A likely reason is dataset composition: openwebtext is dominated by news/political style text, so feature usage is less diverse.
  \item peoples\_speech has broader spoken-language variability, activating a wider portion of the dictionary.
\end{itemize}
\end{frame}

\begin{frame}{Architectural Shift: Why Move Beyond Vanilla SAE?}
\begin{center}
{\scriptsize
\begin{tabular}{>{\raggedright\arraybackslash}p{0.30\textwidth} >{\centering\arraybackslash}p{0.24\textwidth} >{\centering\arraybackslash}p{0.24\textwidth}}
\toprule
\textbf{Metric} & \textbf{Vanilla SAE} & \textbf{FC\_SAE} \\
\midrule
MSE (reconstruction) & 0.1264 & 0.1597 \\
Cosine similarity & 0.9368 & 0.9213 \\
Explained variance & 0.8736 & 0.8421 \\
Feature density & 1.20\% & 3.48\% \\
Dead features & 4716 (38.38\%) & 13 (0.21\%) \\
Validation loss & 0.2575 & 0.2339 \\
\bottomrule
\end{tabular}
}
\end{center}

\vspace{0.25em}
\normalsize
\textbf{Why we moved to FC\_SAE :}
\begin{itemize}
  \item Vanilla SAE has better reconstruction, but severe dead-feature collapse.
  \item FC\_SAE enforces feature-level allocation, sharply improving utilization.
  \item We trade a little reconstruction for a far healthier dictionary for interpretability.
\end{itemize}
\end{frame}

\begin{frame}{Results: Reconstruction/Interpretability Balance}
\begin{itemize}
  \item Too little sparsity: strong reconstruction, weaker disentanglement
  \item Too much sparsity: cleaner features, higher information loss
  \item Practical target: lowest error with stable semantic features
\end{itemize}
\end{frame}


\begin{frame}{Interpretation Protocol (Survey + Ng Notes)}
\begin{itemize}
  \item Collect top-activating tokens/sentences per feature
  \item Assign tentative semantic labels from activation evidence
  \item Validate with counterexamples and cross-context checks
  \item Flag brittle/polysemantic features for retuning
\end{itemize}
\begin{block}{Intuition}
\vspace{2pt}
Interpretable hidden units correspond to sparse non-linear features of input space.
\end{block}
\end{frame}
